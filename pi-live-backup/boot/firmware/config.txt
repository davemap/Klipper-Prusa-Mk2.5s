# --- Crash capture: ramoops/pstore (added 2026-07-05, crash diagnosis) ---
# Reserves a DRAM region so the kernel ring buffer (oops/panic/watchdog reset)
# survives a WARM reboot. Takes effect on NEXT reboot. Read it with:
#   sudo cat /sys/fs/pstore/dmesg-ramoops-0    (right after next boot)
#   ls /var/lib/systemd/pstore/                (systemd archives crash dumps here)
#   journalctl -b 0 -k                         (this-boot kernel log; now persistent)
# CAVEAT: only captures if DRAM survived (kernel panic / soft-lock / watchdog reset).
# A true power BROWNOUT wipes DRAM -> nothing captured; use the FTDI serial console.
dtoverlay=ramoops
